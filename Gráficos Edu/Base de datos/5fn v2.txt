
Table roles {
  id_rol int [pk, increment]
  nombre varchar(50) [not null, unique]
}


Table usuario_rol {
  id_usuario int
  id_rol int

  indexes {
    (id_usuario, id_rol) [pk]
  }
}

Table usuarios {
  id_usuario int [pk, increment]
  email varchar(100) [not null, unique]
  password_hash varchar(255) [not null]
  consentimiento boolean [not null]
  activo boolean [default: true]
  fecha_creacion timestamp [default: `now()`]
}


Table personas {
  id_persona int [pk, increment]
  rut varchar(20) [unique]
  nombre varchar(100)
  apellido varchar(100)
  fecha_nacimiento date
  telefono varchar(20)
}


Ref: usuarios.id_usuario - personas.id_persona

Table medicos {
  id_medico int [pk, increment]
  id_persona int [not null]
}

Table especialidades {
  id_especialidad int [pk, increment]
  nombre varchar(100) [unique]
}


Table medico_especialidad {
  id_medico int
  id_especialidad int

  indexes {
    (id_medico, id_especialidad) [pk]
  }
}

Table pacientes {
  id_paciente int [pk, increment]
  id_persona int [not null]
  numero_ficha varchar(50) [unique]
}

Table instituciones {
  id_institucion int [pk, increment]
  nombre varchar(150)
  direccion varchar(200)
}

Table paciente_institucion {
  id_paciente int
  id_institucion int
  fecha_ingreso date

  indexes {
    (id_paciente, id_institucion, fecha_ingreso) [pk]
  }
}

Table diagnosticos {
  id_diagnostico int [pk, increment]
  nombre varchar(150) [not null, unique]
}

Table registros_clinicos {
  id_registro int [pk, increment]
  id_paciente int [not null]
  id_medico int [not null]
  id_diagnostico int [not null]
  fecha timestamp [default: `now()`]
  observaciones text
}

Table tratamientos {
  id_tratamiento int [pk, increment]
  nombre varchar(100)
  descripcion text
}

Table registro_tratamiento {
  id_registro int
  id_tratamiento int
  dosis varchar(50)
  frecuencia varchar(50)

  indexes {
    (id_registro, id_tratamiento) [pk]
  }
}
